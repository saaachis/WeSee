<script>

const video = document.getElementById("videoElement");
const captureButton = document.getElementById("captureButton");

if (navigator.mediaDevices && navigator.mediaDevices.getUserMedia) {
  navigator.mediaDevices.getUserMedia({ video: true })
    .then(stream => {
      video.srcObject = stream;
      captureButton.addEventListener("click", captureImage);
    })
    .catch(error => {
      console.error("Error accessing camera:", error);
    });
} else {
  console.warn("getUserMedia not supported!");
}

function captureImage() {
  const canvas = document.createElement("canvas");
  canvas.width = video.videoWidth;
  canvas.height = video.videoHeight;
  const context = canvas.getContext("2d");
  context.drawImage(video, 0, 0);

  // Use a timestamp to generate a unique filename for each image capture
  const timestamp = new Date().toISOString().replace(/[-:.]/g, ""); // Removes unwanted characters
  const filename = `camera_image_${timestamp}.jpg`; // Constructs filename with timestamp
  
  const link = document.createElement("a");
  link.href = canvas.toDataURL("image/jpeg"); // Change format if needed (e.g., "image/png")
  link.download = filename; // Use the unique filename
  link.click();
}



</script>
